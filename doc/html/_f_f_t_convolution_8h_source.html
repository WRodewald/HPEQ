90 {

111 template<unsigned int MaxSize>
112 inline void FFTConvolution<MaxSize>::onImpulseResponseUpdated()
113 {
114  unsigned int size = getIR()->getSize();
115  size = IRTools::nextPow2(size);
116 
117  assert(size <= MaxSize);
118 
119  unsigned int requiredOrder = IRTools::staticLog2(size) + 1;
120 
121  unsigned int usedOrder = std::max(requiredOrder, MinOrder);
122  assert(requiredOrder <= MaxOrder);
123 
124  currentFFTOrder = usedOrder;
125 
126  currentFFTSize = std::pow(2, currentFFTOrder - 1);
127 
128  /*
129  Some background:
130  We have a IR of size N=2^n, lets say 16. We want to use the next possible FFT order, so 32. A convolution
131  will spit out (N+M-1) samples, so we can collect 17 samples (2^n+1) befor we need to run the FFT. For simplicity, we can
132  ignore that last sample and just work with 2^n.
133  */
134 
135  numQueuedSamples = 0;
136  outputQueue[0].clear();
137  outputQueue[1].clear();
138  secondaryOutputQueue[0].clear();
139  secondaryOutputQueue[1].clear();
140 
141  // add queue to have enough output for 2^n samples
142  outputQueue[0].push(0, currentFFTSize);
143  outputQueue[1].push(0, currentFFTSize);
144  secondaryOutputQueue[0].push(0, currentFFTSize);
145  secondaryOutputQueue[1].push(0, currentFFTSize);
146 
147  updateKernelFFT();
148 }

168 template<unsigned int MaxSize>
169 inline void FFTConvolution<MaxSize>::performConvolution()
170 {
171  // zero pad input
172  for (int i = currentFFTSize; i < (2 * currentFFTSize); i++)
173  {
174  audioFFTs[0][i] = audioFFTs[1][i] = 0;
175  }
176 
177  // to frequency domain
178  fftEngines[currentFFTOrder]->performFFTInPlace(audioFFTs[0].data());
179  fftEngines[currentFFTOrder]->performFFTInPlace(audioFFTs[1].data());
180 
181  // per sample multiplication
182  for (int i = 0; i < (2 * currentFFTSize); i++)
183  {
184  audioFFTs[0][i] *= kernelFFTs[0][i];
185  audioFFTs[1][i] *= kernelFFTs[1][i];
186  }
187 
188  // back to time domain
189  fftEngines[currentFFTOrder]->performIFFTInPlace(audioFFTs[0].data());
190  fftEngines[currentFFTOrder]->performIFFTInPlace(audioFFTs[1].data());
191 
192 
193  // cache the first half of the samples directly to the output
194  // cache the second half to the queue for later use
195  for (int i = 0; i < currentFFTSize; i++)
196  {
197  // write first half and secondary queue to output queue
198  outputQueue[0].push(audioFFTs[0][i].real() + secondaryOutputQueue[0].pull());
199  outputQueue[1].push(audioFFTs[1][i].real() + secondaryOutputQueue[1].pull());
200 
201  // write second half to secondary queue for later use
202  secondaryOutputQueue[0].push(audioFFTs[0][i + currentFFTSize].real());
203  secondaryOutputQueue[1].push(audioFFTs[1][i + currentFFTSize].real());
204  }
205 
206  numQueuedSamples = 0;
207 }
