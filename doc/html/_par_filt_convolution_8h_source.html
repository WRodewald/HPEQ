245 template<typename T, unsigned int MaxOrder>
246 inline T FIRFilter<T, MaxOrder>::tick(const T & input)
247 {
248  auto out = this->coeffs.b[0] * input + this->z[0];
249  for (int i = 0; i < (static_cast<int>(curOrder)- 1); i++)
250  {
251  this->z[i] = this->coeffs.b[i + 1] * input + z[i + 1];
252  }
253  z[curOrder - 1] = this->coeffs.b[curOrder] * input;
254  return out;
255 }
