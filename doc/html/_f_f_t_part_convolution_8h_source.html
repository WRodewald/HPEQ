114

135 template<unsigned int MaxSize>
136 inline void FFTPartConvolution<MaxSize>::onDataUpdate()
137 {
138  unsigned int size = getData()->getSize();
139  size = IRTools::nextPow2(size);
140 
141  assert(size <= MaxSize);
142 
143  // now calculating the size of parititons
144  unsigned int partSize = size >> requestedPartOrder;
145 
146  unsigned int requiredFFTOrder = IRTools::staticLog2(partSize) + 1;
147 
148  unsigned int usedOrder = std::max(requiredFFTOrder, MinOrder);
149  assert(requiredFFTOrder <= MaxOrder);
150 
151  currentFFTOrder = usedOrder;
152 
153  currentFFTSize = std::pow(2, currentFFTOrder - 1);
154 
155  numPartitions = std::max(size / currentFFTSize, static_cast<unsigned int>(1));
156 
157  /*
158  Some background:
159  We have a IR of size N=2^n, lets say 16. We want to use the next possible FFT order, so 32. A convolution
160  will spit out (N+M-1) samples, so we can collect 17 samples (2^n+1) befor we need to run the FFT. For simplicity, we can
161  ignore that last sample and just work with 2^n.
162  */
163 
164  // clear buffers, prepare queues
165  for (auto c : { 0,1 })
166  {
167  audioInput[c].fill(0);
168  partitionFFTCache[c].fill(0);
169  outputQueue[c].clear();
170  outputQueue[c].push(0, currentFFTSize);
171 
172  secondaryOutputQueue[c].clear();
173  secondaryOutputQueue[c].push(0, currentFFTSize);
174  }
175 
176  numQueuedSamples = 0;
177  currentPartition = 0;
178 
179 
180  updateKernelFFT();
181 }

197 inline void FFTPartConvolution<MaxSize>::updateKernelFFT()
198 {
204  // set order, just to be sure
205  currentFFTOrder = IRTools::staticLog2(currentFFTSize) + 1;
206  for (int c : {0, 1})
207  {
208  kernelFFTs[c].fill(0);
209 
210  auto buffer = getData()->getChannel(c);
211 
212  for (int p = 0; p < numPartitions; p++)
213  {
214  unsigned int irBufferOffset = p * currentFFTSize;
215  unsigned int fftBufferOffset = 2 * irBufferOffset;
216 
217  unsigned int irMaxPos = std::min(getData()->getSize() - irBufferOffset, currentFFTSize);
218  for (int i = 0; i < irMaxPos; i++)
219  {
220  kernelFFTs[c][i + fftBufferOffset] = buffer[i + irBufferOffset];
221  }
222 
223  fftEngines[currentFFTOrder]->performFFTInPlace(&kernelFFTs[c][fftBufferOffset]);
224  }

228 template<unsigned int MaxSize>
229 inline void FFTPartConvolution<MaxSize>::performConvolution()
230 {
231  /*
232  For every performConvolution call, we need to:
233  - perform FFT of the new input
234  - for every partition
235  - complex multiplication
236  - sum up all partiotions
237  - ifft
238  - feed queues
239 
240  Note on the partitionFFTCache: We handle it as a ring buffer,
241 
242  */
243 
244 
245  // zero pad input
246  for (int i = currentFFTSize; i < (2 * currentFFTSize); i++)
247  {
248  audioInput[0][i] = audioInput[1][i] = 0;
249  }
250 
251  auto currentPartitionChaceOffset = currentPartition * (2 * currentFFTSize);
252 
253  // to frequency domain
254  fftEngines[currentFFTOrder]->performFFT(audioInput[0].data(), &partitionFFTCache[0][currentPartitionChaceOffset]);
255  fftEngines[currentFFTOrder]->performFFT(audioInput[1].data(), &partitionFFTCache[1][currentPartitionChaceOffset]);
256 
257  // first partition out of loop so that we don't need to flush the buffer first
258  for (int i = 0; i < (2*currentFFTSize); i++)
259  {
260  audioInput[0][i] = partitionFFTCache[0][currentPartitionChaceOffset + i] * kernelFFTs[0][i];
261  audioInput[1][i] = partitionFFTCache[1][currentPartitionChaceOffset + i] * kernelFFTs[1][i];
262  }
263 
264  // per parition
265  for (int p = 1; p < numPartitions; p++)
266  {
267  auto kernelOffset = 2 * currentFFTSize * p;
268  auto cacheOffset = 2 * currentFFTSize * ((currentPartition + numPartitions - p) % numPartitions);
269  for (int i = 0; i < (2 * currentFFTSize); i++)
270  {
271  audioInput[0][i] += partitionFFTCache[0][cacheOffset + i] * kernelFFTs[0][kernelOffset + i];
272  audioInput[1][i] += partitionFFTCache[1][cacheOffset + i] * kernelFFTs[1][kernelOffset + i];
273  }
274  }
275 
276  // back to time domain
277  fftEngines[currentFFTOrder]->performIFFTInPlace(audioInput[0].data());
278  fftEngines[currentFFTOrder]->performIFFTInPlace(audioInput[1].data());
279 
280 
281  // cache the first half of the samples directly to the output
282  // cache the second half to the queue for later use
283  for (int i = 0; i < currentFFTSize; i++)
284  {
285  // write first half and secondary queue to output queue
286  outputQueue[0].push(audioInput[0][i].real() + secondaryOutputQueue[0].pull());
287  outputQueue[1].push(audioInput[1][i].real() + secondaryOutputQueue[1].pull());
288 
289  // write second half to secondary queue for later use
290  secondaryOutputQueue[0].push(audioInput[0][i + currentFFTSize].real());
291  secondaryOutputQueue[1].push(audioInput[1][i + currentFFTSize].real());
292  }
293 
294  // increment artition
295  currentPartition = (currentPartition + 1) % numPartitions;
296  numQueuedSamples = 0;
297 }
